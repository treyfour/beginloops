## Suggested next step
Create a custom React hook `hooks/useHeartbeat.ts` that polls your heartbeat endpoint on an interval and returns live system status — then render a small status badge in the UI:

```ts
// hooks/useHeartbeat.ts
import { useEffect, useState } from "react";

interface HeartbeatResult {
  status: "ok" | "degraded" | "loading";
  checks: Record<string, string>;
  timestamp: string | null;
}

export function useHeartbeat(intervalMs = 30_000): HeartbeatResult {
  const [result, setResult] = useState<HeartbeatResult>({
    status: "loading",
    checks: {},
    timestamp: null,
  });

  useEffect(() => {
    const poll = async () => {
      try {
        const res = await fetch("/api/cron/heartbeat");
        const data = await res.json();
        setResult({ status: data.status, checks: data.checks, timestamp: data.timestamp });
      } catch {
        setResult((prev) => ({ ...prev, status: "degraded" }));
      }
    };

    poll();
    const id = setInterval(poll, intervalMs);
    return () => clearInterval(id);
  }, [intervalMs]);

  return result;
}
```

Then in any page or layout component:
```tsx
const { status, checks } = useHeartbeat();
// render a green/red dot based on status
```

Why it matters: this is the **hooks** pillar of the learning rotation — custom hooks let you pull async, side-effectful logic out of components and reuse it anywhere. The pattern of "poll an endpoint, store state, auto-cleanup on unmount" appears constantly in dashboards, status pages, and real-time UIs. The `clearInterval` return from `useEffect` is also a key concept: it prevents memory leaks when the component unmounts.
