## Suggested next step
Build a custom React hook `useHeartbeat` in `app/hooks/useHeartbeat.ts` that polls your `/api/cron/heartbeat` endpoint from the browser — this connects your backend infrastructure to the frontend and teaches you the **hooks** pattern central to modern React.

```ts
// app/hooks/useHeartbeat.ts
"use client";
import { useEffect, useState } from "react";

interface HeartbeatResult {
  status: "ok" | "degraded" | "loading" | "error";
  timestamp: string | null;
  checks: { webhookEndpoint: "ok" | "unreachable" } | null;
}

export function useHeartbeat(intervalMs = 30_000): HeartbeatResult {
  const [result, setResult] = useState<HeartbeatResult>({
    status: "loading",
    timestamp: null,
    checks: null,
  });

  useEffect(() => {
    async function poll() {
      try {
        const res = await fetch("/api/cron/heartbeat");
        const data = await res.json();
        setResult({ status: data.status, timestamp: data.timestamp, checks: data.checks });
      } catch {
        setResult((prev) => ({ ...prev, status: "error" }));
      }
    }

    poll();
    const id = setInterval(poll, intervalMs);
    return () => clearInterval(id); // cleanup prevents memory leaks
  }, [intervalMs]);

  return result;
}
```

Then use it in any Server or Client component:

```tsx
// app/page.tsx (add "use client" at the top)
import { useHeartbeat } from "./hooks/useHeartbeat";

export default function Home() {
  const { status, timestamp } = useHeartbeat();
  return <p>Webhook endpoint: {status} — last checked {timestamp ?? "…"}</p>;
}
```

**Why this matters:** the `useEffect` cleanup (`return () => clearInterval(id)`) is the canonical pattern for managing side effects in React — missing it causes a memory leak where the interval keeps firing even after the component unmounts. This hook also shows how a custom hook is just a function that calls other hooks; it can be composed, tested, and reused anywhere.
