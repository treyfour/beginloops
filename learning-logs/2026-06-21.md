Add HMAC-SHA256 signature verification to `app/api/webhooks/route.ts` so you can confirm incoming requests are genuinely from GitHub (not spoofed):

```ts
// app/api/webhooks/route.ts — add at the top of POST()
import { createHmac, timingSafeEqual } from "crypto";

const secret = process.env.WEBHOOK_SECRET ?? "";

async function verifySignature(req: NextRequest, rawBody: string): Promise<boolean> {
  const sig = req.headers.get("x-hub-signature-256") ?? "";
  const expected = "sha256=" + createHmac("sha256", secret).update(rawBody).digest("hex");
  return timingSafeEqual(Buffer.from(sig), Buffer.from(expected));
}

export async function POST(req: NextRequest) {
  const rawBody = await req.text();          // read as text FIRST
  if (!(await verifySignature(req, rawBody))) {
    return NextResponse.json({ error: "Invalid signature" }, { status: 401 });
  }
  const body = JSON.parse(rawBody);          // then parse
  // ... rest of handler unchanged
}
```

Why it matters: right now any HTTP client can POST to your endpoint and your handler will process it. Signature verification uses a shared secret (set as `WEBHOOK_SECRET` in Vercel env vars, and in the GitHub webhook settings) to cryptographically prove the request came from GitHub. `timingSafeEqual` prevents timing attacks. This is the step that makes a webhook handler production-safe.
