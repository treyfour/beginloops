## Suggested next step
Create a custom React hook `app/hooks/useHeartbeat.ts` that polls `/api/cron/heartbeat` every 30 seconds and returns the latest status — then wire it into `app/page.tsx` to show a live "service health" badge.

```ts
// app/hooks/useHeartbeat.ts
"use client";
import { useEffect, useState } from "react";

interface HeartbeatStatus {
  status: string;
  timestamp: string;
}

export function useHeartbeat(intervalMs = 30_000) {
  const [data, setData] = useState<HeartbeatStatus | null>(null);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    async function poll() {
      try {
        const res = await fetch("/api/cron/heartbeat");
        const json = await res.json();
        setData(json);
        setError(null);
      } catch {
        setError("unreachable");
      }
    }

    poll(); // run immediately on mount
    const id = setInterval(poll, intervalMs);
    return () => clearInterval(id); // cleanup on unmount
  }, [intervalMs]);

  return { data, error };
}
```

Then in `app/page.tsx` (mark it `"use client"` at the top):

```tsx
import { useHeartbeat } from "./hooks/useHeartbeat";

export default function Home() {
  const { data, error } = useHeartbeat();
  return (
    <main>
      <p>
        Service:{" "}
        {error ? "❌ unreachable" : data ? `✅ ${data.status}` : "checking…"}
      </p>
    </main>
  );
}
```

**Why this matters:** Custom hooks are the main way React components share stateful logic without duplicating code. The `use` prefix is a convention the linter enforces — it signals that this function can call other hooks. The `return () => clearInterval(id)` cleanup function is critical: without it, every time the component re-mounts, a new interval fires and the old one keeps running forever (a classic memory leak). This pattern — fetch on mount, poll on interval, clean up on unmount — appears in nearly every real-time dashboard, status page, and live feed you'll ever build.
